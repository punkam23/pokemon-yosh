package com.ucenfotec.pokemonyosh.model;

import lombok.Setter;

@Setter
@AllArgsConstructor
@NoArgsConstructor
public class Pokemon {
    @JsonProperty("name")
    String name;

    @JsonProperty("type")
    PokemonTypeEnum type;

    @JsonProperty("life")
    double life;

    @JsonProperty("attacks")
    List<Attack> attacks;
}
